<script type="text/javascript">
  function battlefield_players(){
    $('table.battlefield tr.player a.player').click(function(){
      var row = $(this).parents('tr.player')

      if (row.next().is('.profile')) {
        row.next().remove();
        row.removeClass('current')
      } else {
        $.get('/stats.php?id='+row.attr('user_id'), function(data){
          $('table.battlefield tr.current').removeClass('current')
          $('table.battlefield tr.profile').remove()
          row.addClass('current')
          row.after(
            $('<tr class="profile"><td colspan="7"></td></tr>').find('td').html(data).end()
          )
          $('span.alliance:gt(10)').hide(); $('span.alliance:eq(10)').after( $('<a href="#">...</a>').click(function(){ $(this).remove(); $('span.alliance').show(); return false }) )
          profile_buddy_change()
        })
      }

      return false
    })
  }
  function battlefield_nav(){
    $('table.battlefield tr.nav a').click(function(){
      $.get($(this).attr('href'), function(data){
        $('table.battlefield tr:first').nextAll().remove()
        $('table.battlefield tr:first').after(data)

        battlefield_players()
        battlefield_nav()
      })
      return false
    })
  }
  function profile_buddy_change(){
    $('select[name=buddy_type]').change(function(){
      var select = $(this)
      $.post('/addbuddy.php', select.parents('form:first').serialize(), function(){
        var img = select.prev()
        var val = select.find('option:selected').attr('value')

        img.add( img.parents('tr.profile').prev().find('td > img.buddy_type') ).each(function(){
          var i = $(this)
          i.attr('src', img.attr('src').replace(/icon_.*\.gif/, 'icon_' + val + '.gif'))
          if (val == 'none')
            i.hide()
          else
            i.show()
        })
      })
    })
  }
  battlefield_players()
  battlefield_nav()
</script>
